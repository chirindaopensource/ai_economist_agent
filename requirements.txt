#   source .venv/bin/activate          # Linux / macOS

#   # Install all dependencies at exact pinned versions:
#   pip install --require-hashes --no-deps -r requirements.txt
#   # OR, for environments that cannot use --require-hashes:
#   pip install -r requirements.txt
#
# Pinning Strategy
# ----------------
#   All packages are pinned with == (exact version) to guarantee byte-for-byte
#   reproducibility of the numerical results reported in the paper. The study
#   enforces deterministic decoding (temperature=0, seed=42) and deterministic
#   PRNG draws (numpy.random.default_rng(seed=42)); any silent float-precision
#   change induced by a library upgrade would silently corrupt reproducibility.
#
#   Transitive dependencies are pinned explicitly (not left implicit) because
#   pip's dependency resolver is not guaranteed to re-select the same transitive
#   versions across pip releases or platform configurations. Explicit pinning of
#   the full closure is the only method that guarantees hermetic reproduction.
#
# Dependency Tiers
# ----------------
#   TIER 1 : Direct dependencies — imported by name in ai_economist_agent.py
#   TIER 2 : Transitive dependencies — required by Tier-1 packages; pinned
#             explicitly for hermeticity
#
# External Runtime Requirements (NOT pip-installable; install separately)
# -----------------------------------------------------------------------
#   Ollama >= 0.3.0   : Local LLM inference server exposing the OpenAI-
#                        compatible endpoint at http://localhost:11434/v1.
#                        Install: https://ollama.com/download
#                        Pull model: ollama pull llama3.1:8b
#   Neo4j  >= 5.x     : Graph database server. The Python neo4j driver
#                        (pinned below) connects to a running Neo4j instance.
#                        Install: https://neo4j.com/download/
#                        Default URI: bolt://localhost:7687

# ── Scientific Computing Stack ─────────────────────────────────────────────────

# NumPy: Foundation for all vectorised mathematical operations.
# Used for: PRNG initialisation (numpy.random.default_rng(seed=42)),
# matrix algebra (Blanchard-Kahn eigenvalue decomposition), state-space
# simulation, and all float64 array operations mandated by
# computational_solver_parameters.numerical_precision = "float64".
# NOTE: NumPy 2.x removes legacy type aliases (np.bool, np.int, np.float).
# The codebase must NOT use these deprecated aliases; use np.bool_, np.int64,
# np.float64 exclusively. Validated on 2.4.4.
# Disciplines: 3, 4, 5, 6, 9
numpy==2.4.4

# pandas: Tabular data management for the evidence corpus (2,500 rows),
# model specification catalog (420 specs), and temporal fact tables.
# Used for: schema validation via pandas.api.types (ptypes), DataFrame
# construction for evidence retrieval ranking, and ModelOutputPoint
# time-series extraction across the 12-step horizon.
# Disciplines: 3, 8, 9
pandas==3.0.2

# SciPy: Advanced scientific computation.
# Used for:
#   scipy.stats.beta(a=9, b=1)  — Beta distribution for synthetic confidence
#                                  score generation (Discipline 3, Table 6).
#   scipy.linalg.solve_discrete_are — Algebraic Riccati equation for DSGE
#                                      state-space solution (Discipline 4).
#   scipy.linalg.ordqz          — Ordered QZ decomposition for
#                                  Blanchard-Kahn generalised eigenvalue
#                                  verification (Discipline 4).
#   scipy.special.expit         — Logistic sigmoid for PD mapping:
#                                  PD_t = expit(α₀ + α₁·p_CRE + α₂·s_credit)
#                                  (Discipline 6).
# Disciplines: 3, 4, 5, 6
scipy==1.17.1

# ── HTTP Client ────────────────────────────────────────────────────────────────

# requests: Synchronous HTTP client for all LLM agent calls.
# Used for: POST requests to the Ollama OpenAI-compatible endpoint
# (http://localhost:11434/v1/chat/completions) with JSON-serialised
# payloads enforcing temperature=0, seed=42, and max_tokens=2048.
# All six agent calls (Planner, Model-Selection, three Report Generators,
# Judge) route through requests.Session for connection reuse.
# Discipline: 1, 7, 8
requests==2.33.1

# ── System Introspection ────────────────────────────────────────────────────────

# psutil: Cross-platform system and process utilities.
# Used for: runtime environment fingerprinting (CPU count, RAM, GPU memory)
# required by the cryptographic audit hash of the execution environment
# (hashlib + platform + psutil → SHA-256 environment digest for regulatory
# auditability). Also used for monitoring memory pressure during sequential
# agentic KV-cache calls (Discipline 1).
# Disciplines: 1, 9
psutil==7.2.2

# ── Graph Database Driver ───────────────────────────────────────────────────────

# neo4j: Official Neo4j Python driver (Bolt protocol).
# Used for: all Knowledge Graph operations — atomic graph reset (MATCH (n)
# DETACH DELETE n), node and relationship creation, pre-execution Cypher
# path traversal (Scenario→VariableConcept→ModelSpecification→
# ImplementationFunction), atomic ModelRun/ModelOutputPoint write-back,
# and post-execution diagnostic path inspection.
# Imports used in source: Driver, Transaction, Record, neo4j.exceptions,
#   neo4j.graph.Path, neo4j.graph.Node, neo4j.graph.Relationship.
# NOTE: neo4j 6.x driver requires Neo4j server >= 5.x. The driver exposes
# read-committed isolation semantics by default, satisfying the graph-state
# isolation requirement of Discipline 2.
# Discipline: 2
neo4j==6.2.0

# ── Data Validation ────────────────────────────────────────────────────────────

# pydantic: Runtime data validation and settings management (v2 API).
# Used for: strict JSON schema enforcement on all LLM agent outputs.
# Every agent response (Planner JSON, Model-Selection execution request
# tuple, Judge scoring JSON) is parsed against a Pydantic BaseModel before
# being passed downstream in the DAG. Prevents type coercion errors and
# enforces the structured execution request tuple r_s = (spec_id,
# execution_tool, expected_outputs).
# NOTE: This pin is for Pydantic v2 (2.x). The v2 API uses field_validator
# (not the v1 @validator decorator). The codebase must use v2 idioms.
# Discipline: 8
pydantic==2.13.4
